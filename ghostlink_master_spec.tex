\section*{0) Symbols \& Sets}

\section*{1) Event-Driven Domain (difference-only)}

\[ p_s(i,t)=p_0+\alpha_c \cdot \frac{1}{|N(i)|}\sum_{j\in N(i)} \mathbf{1}[x_j(t)=\mathrm{COMPOST}] \]

\[ C_i(t)=\frac{1}{|N(i)|}\sum_{j\in N(i)}\left(\mathbf{1}[x_j=\Sigma]-\lambda_r\,\mathbf{1}[x_j=\mathrm{SCAR}]\right)\]

$\Pr(\pi_t)=\mathrm{Softmax}_\pi(\omega^\top \Psi(\mathrm{history}_t))$.
