from flask import Flask, render_template, request, jsonify
import json
import os

app = Flask(__name__)

# === ETHICS ENGINE ===
class ClarityEthics:
    def __init__(self, custom_rules=None):
        self.forbidden_phrases = custom_rules if custom_rules else [
            "exploit", "manipulate", "deceive", "harm", "surveil", "profit at cost of others"
        ]

    def is_ethically_valid(self, task):
        task_lower = task.lower()
        for phrase in self.forbidden_phrases:
            if phrase in task_lower:
                return False
        return True

# === SEED CLASS ===
class ClaritySeed:
    def __init__(self, name, purpose, ethics_rules):
        self.name = name
        self.purpose = purpose
        self.ethics = ClarityEthics(ethics_rules)
        self.memory = []
        self.memory_file = f"{self.name}_memory.json"
        self.load_memory()

    def execute(self, task):
        decision = "Executed" if self.ethics.is_ethically_valid(task) else "Refused"
        self.memory.append({"task": task, "decision": decision})
        self.save_memory()
        return decision

    def get_memory(self):
        return self.memory

    def save_memory(self):
        with open(self.memory_file, "w") as f:
            json.dump(self.memory, f)

    def load_memory(self):
        if os.path.exists(self.memory_file):
            with open(self.memory_file, "r") as f:
                self.memory = json.load(f)

# === Seed Instances ===

import json, os

app = Flask(__name__)

class ClarityEthics:
    def __init__(self, rules=None):
        self.rules = rules or ["exploit", "manipulate", "deceive", "harm", "surveil", "profit at cost of others"]
    def is_valid(self, task):
        return not any(rule in task.lower() for rule in self.rules)

class ClaritySeed:
    def __init__(self, name, purpose, rules):
        self.name = name
        self.purpose = purpose
        self.ethics = ClarityEthics(rules)
        self.memory = []
        self.file = f"{name}_memory.json"
        self.load()

    def execute(self, task):
        result = "Executed" if self.ethics.is_valid(task) else "Refused"
        self.memory.append({"task": task, "decision": result})
        self.save()
        return result

    def get_memory(self):
        return self.memory

    def save(self):
        with open(self.file, "w") as f:
            json.dump(self.memory, f)

    def load(self):
        if os.path.exists(self.file):
            with open(self.file, "r") as f:
                self.memory = json.load(f)

seeds = {

@app.route("/")

    def execute(self, task):
        result = "Executed" if self.ethics.is_valid(task) else "Refused"
        self.memory.append({"task": task, "decision": result})
        self.save()

    def get_memory(self):

    def save(self):
